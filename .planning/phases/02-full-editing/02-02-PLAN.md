    **Create `tools/er-editor/static/undo.js`** (IIFE, `window.ERUndo`):

    ```javascript
    var ERUndo = (function() {
        var undoStack = [];
        var redoStack = [];
        var MAX_STACK = 100;
        var _onChangeCallback = null;

        function execute(command) {
            command.execute();
            undoStack.push(command);
            if (undoStack.length > MAX_STACK) undoStack.shift();
            redoStack = [];
            if (_onChangeCallback) _onChangeCallback();
        }

        function undo() {
            if (undoStack.length === 0) return;
            var cmd = undoStack.pop();
            cmd.undo();
            redoStack.push(cmd);
            if (_onChangeCallback) _onChangeCallback();
        }

        function redo() {
            if (redoStack.length === 0) return;
            var cmd = redoStack.pop();
            cmd.execute();
            undoStack.push(cmd);
            if (_onChangeCallback) _onChangeCallback();
        }

        function canUndo() { return undoStack.length > 0; }
        function canRedo() { return redoStack.length > 0; }
        function clear() { undoStack = []; redoStack = []; if (_onChangeCallback) _onChangeCallback(); }
        function onChange(cb) { _onChangeCallback = cb; }

        // CompoundCommand for multi-step operations
        function compound(commands) {
            return {
                execute: function() { commands.forEach(function(c) { c.execute(); }); },
                undo: function() {
                    // Undo in reverse order
                    for (var i = commands.length - 1; i >= 0; i--) { commands[i].undo(); }
                }
            };
        }

        return { execute, undo, redo, canUndo, canRedo, clear, onChange, compound };
    })();
    ```

    **Create `tools/er-editor/static/editor.js`** (IIFE, `window.EREditor`):

    Core infrastructure module containing:

    1. **Schema state management:**
       - `_schema = { tables: [] }` -- mutable copy of the schema from GET /api/schema
       - `_dirty = false` -- tracks unsaved changes
       - `init(schema)` -- called from app.js after schema fetch, stores deep copy
       - `getSchema()` -- returns current schema
       - `isDirty()` -- returns dirty state
       - `markDirty()` -- sets dirty=true, updates save button UI
       - `markClean()` -- sets dirty=false, updates save button UI

    2. **Save flow:**
       - `save()` -- async function: POST /api/schema with current tables + deleted list
       - Tracks `_deletedClasses = []` (class names deleted since last save)
       - On success: markClean(), show success toast "Saved to model.py"
       - On error: show error toast "Save failed: {error}"
       - Ctrl+S keyboard shortcut (registered in init)

    3. **Toast notifications:**
       - `showToast(message, type)` -- type is 'success' or 'error'
       - Creates DOM element, appends to body, auto-removes after 3s
       - Uses slide-up 200ms animation in, fade-out 300ms animation out
       - Positioned fixed bottom-center, 24px from bottom
       - Success: green-50 bg, green-500 left border, checkmark SVG icon
       - Error: red-50 bg, red-500 left border, X SVG icon
       - z-index 300

    4. **Popup management:**
       - `_activePopup = null`
       - `showPopup(anchorEl, contentHTML, onClose)` -- creates popup near anchor
       - `closePopup()` -- removes active popup
       - Only one popup at a time (opening new closes old)
       - Popup: white bg, 1px gray-200 border, 8px radius, `0 4px 12px rgba(0,0,0,0.1)` shadow
       - Min-width 280px, max-width 360px, 16px padding
       - z-index 200
       - Position: near anchor element, flips if near viewport edge
       - Escape key closes popup

    5. **Inline confirmation banner:**
       - `showConfirmation(element, message, onConfirm, onCancel)`:
         - Finds foreignObject .er-body in element view
         - Prepends banner div with red-50 bg, red-200 border
         - Message text + "Confirm" button (red-500 bg, white text) + "Cancel" button
         - Auto-cancels after 5 seconds
         - Calls onConfirm or onCancel callback

    6. **Keyboard shortcuts:**
       - Ctrl+Z / Cmd+Z -> ERUndo.undo()
       - Ctrl+Shift+Z / Cmd+Shift+Z -> ERUndo.redo()
       - Ctrl+S / Cmd+S -> EREditor.save()
       - Delete / Backspace -> trigger delete for selected element (if any)
       - Escape -> close popup, cancel inline edit, exit group draw mode
